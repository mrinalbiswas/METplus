152 def parse_launch_args(args, logger, base_confs=METPLUS_BASE_CONFS):
153  """!Parsed arguments to scripts that launch the METplus system.
154 
155  This is the argument parser for the config_metplus.py
156  script. It parses the arguments (in args).
157  If something goes wrong, this function calls
158  sys.exit(1) or sys.exit(2).
159 
160  Options:
161  * section.variable=value --- set this value in this section, no matter what
162  * /path/to/file.conf --- read this conf file after the default conf files.
163 
164  Later conf files override earlier ones. The conf files read in
165  are:
166  * metplus.conf
167 
168  @param args the script arguments, after script-specific ones are removed
169  @param logger a logging.Logger for log messages"""
170 
171  parm = os.path.realpath(PARM_BASE)
172 
173  # Files in this list, that don't exist or are empty,
174  # will be silently ignored.
175  # infiles=[ os.path.join(parm, 'metplus.conf'),
176  # os.path.join(parm, 'metplus.override.conf')
177  # ]
178  infiles = list()
179  for base_conf in base_confs:
180  infiles.append(base_conf)
181 
182  moreopt = collections.defaultdict(dict)
183 
184  if args is None:
185  return (parm, infiles, moreopt)
186 
187  # Now look for any option and conf file arguments:
188  bad = False
189  for iarg in range(len(args)):
190  m = re.match(r'''(?x)
191  (?P<section>[a-zA-Z][a-zA-Z0-9_]*)
192  \.(?P<option>[^=]+)
193  =(?P<value>.*)$''', args[iarg])
194  if m:
195  logger.info('Set [%s] %s = %s' % (
196  m.group('section'), m.group('option'),
197  repr(m.group('value'))))
198  moreopt[m.group('section')][m.group('option')] = m.group('value')
199  elif os.path.exists(args[iarg]):
200  infiles.append(args[iarg])
201  elif os.path.exists(os.path.join(parm, args[iarg])):
202  infiles.append(os.path.join(parm, args[iarg]))
203  else:
204  bad = True
205  # if OS separator character (/ or \) if in argument, assume it
206  # is supposed to be a path but the file is not found
207  if os.sep in args[iarg]:
208  logger.error(f"Configuration file not found: {args[iarg]}")
209  else:
210  logger.error(f"Invalid argument: {args[iarg]}")
211  if bad:
212  sys.exit(2)

215  if not os.path.exists(file):

359  if log_timestamp_template:
360  # Note: strftime appears to handle if log_timestamp_template
361  # is a string ie. 'blah' and not a valid set of % directives %Y%m%d,
362  # it does return the string 'blah', instead of crashing.
363  # However, I'm still going to test for a valid % directive and

468  # not create it.

650  # override get methods to perform additional error checking
651  def getraw(self, sec, opt, default='', count=0):
652  """ parse parameter and replace any existing parameters
653  referenced with the value (looking in same section, then
654  config, dir, and os environment)
655  returns raw string, preserving {valid?fmt=%Y} blocks
656  Args:
657  @param sec: Section in the conf file to look for variable
658  @param opt: Variable to interpret
659  @param default: Default value to use if config is not set
660  @param count: Counter used to stop recursion to prevent infinite
661  Returns:
662  Raw string or empty string if function calls itself too many times
663  """
664  count = count + 1
665  if count >= 10:
666  return ''
667 
668  # if requested section is in the list of sections that are no longer used
669  # look in the [config] section for the variable
670  if sec in self.OLD_SECTIONS:
671  sec = 'config'
672 
673  in_template = super().getraw(sec, opt, default)
674  out_template = ""
675  in_brackets = False
676  for index, character in enumerate(in_template):
677  if character == "{":
678  in_brackets = True
679  start_idx = index
680  elif character == "}":
681  var_name = in_template[start_idx+1:index]
682  var = None
683  if self.has_option(sec, var_name):
684  var = self.getraw(sec, var_name, default, count)
685  elif self.has_option('config', var_name):
686  var = self.getraw('config', var_name, default, count)
687  elif var_name[0:3] == "ENV":
688  var = os.environ.get(var_name[4:-1])
689 
690  if var is None:
691  out_template += in_template[start_idx:index+1]
692  else:
693  out_template += var
694  in_brackets = False
695  elif not in_brackets:
696  out_template += character
697 
698  # replace double slash in path to single slash
699  return out_template.replace('//', '/')
