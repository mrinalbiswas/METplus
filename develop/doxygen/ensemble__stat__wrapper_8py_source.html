251  def find_model_members(self, time_info):
252  """! Finds the model member files to compare
253  Args:
254  @param time_info dictionary containing timing information
255  @rtype string
256  @return Returns a list of the paths to the ensemble model files
257  """
258  model_dir = self.c_dict['FCST_INPUT_DIR']
259  # used for filling in missing files to ensure ens_thresh check is accurate
260  fake_dir = '/ensemble/member/is/missing'
261 
262  # model_template is a list of 1 or more.
263  ens_members_path = []
264 
265  # get all files that exist
266  for ens_member_template in self.c_dict['FCST_INPUT_TEMPLATE']:
267  member_file = do_string_sub(ens_member_template,
268  **time_info)
269  expected_path = os.path.join(model_dir, member_file)
270 
271  # if wildcard expression, get all files that match
272  if '?' in expected_path or '*' in expected_path:
273  wildcard_files = sorted(glob.glob(expected_path))
274  self.logger.debug('Ensemble members file pattern: {}'
275  .format(expected_path))
276  self.logger.debug('{} members match file pattern'
277  .format(str(len(wildcard_files))))
278 
279  # add files to list of ensemble members
280  for wildcard_file in wildcard_files:
281  ens_members_path.append(wildcard_file)
282  else:
283  # otherwise check if file exists
284  expected_path = util.preprocess_file(expected_path,
285  self.c_dict['FCST_INPUT_DATATYPE'],
286  self.config)
287 
288  # if the file exists, add it to the list
289  if expected_path:
290  ens_members_path.append(expected_path)
291  else:
292  # add relative path to fake dir and add to list
293  ens_members_path.append(os.path.join(fake_dir, member_file))
294  self.logger.warning('Expected ensemble file {} not found'
295  .format(member_file))
296 
297  # if more files found than expected, error and exit
298  if len(ens_members_path) > self.c_dict['N_MEMBERS']:
299  msg = 'Found more files than expected! ' +\
300  'Found {} expected {}. '.format(len(ens_members_path),
301  self.c_dict['N_MEMBERS']) +\
302  'Adjust wildcard expression in [filename_templates] '+\
303  'FCST_ENSEMBLE_STAT_INPUT_TEMPLATE or adjust [config] '+\
304  'ENSEMBLE_STAT_N_MEMBERS. Files found: {}'.format(ens_members_path)
305  self.log_error(msg)
306  self.logger.error("Could not file files in {} for init {} f{} "
307  .format(model_dir, time_info['init_fmt'],
308  str(time_info['lead_hours'])))
309  return False
310  # if fewer files found than expected, warn and add fake files
311  elif len(ens_members_path) < self.c_dict['N_MEMBERS']:
312  msg = 'Found fewer files than expected. '+\
313  'Found {} expected {}.'.format(len(ens_members_path),
314  self.c_dict['N_MEMBERS'])
315  self.logger.warning(msg)
316  # add fake files to list to get correct number of files for ens_thresh
317  diff = self.c_dict['N_MEMBERS'] - len(ens_members_path)
318  self.logger.warning('Adding {} fake files to '.format(str(diff))+\
319  'ensure ens_thresh check is accurate')
320  for _ in range(0, diff, 1):
321  ens_members_path.append(fake_dir)
322 
323  # write file that contains list of ensemble files
324  list_filename = time_info['init_fmt'] + '_' + \
325  str(time_info['lead_hours']) + '_ensemble.txt'
326  return self.write_list_file(list_filename, ens_members_path)
