443  def get_all_field_info(self, var_list, data_type):
444  """!Get field info based on data type"""
445 
446  field_list = []
447  for var_info in var_list:
448  if data_type == 'FCST':
449  level = var_info['fcst_level']
450  thresh = var_info['fcst_thresh']
451  name = var_info['fcst_name']
452  extra = var_info['fcst_extra']
453  elif data_type == 'OBS':
454  level = var_info['obs_level']
455  thresh = var_info['obs_thresh']
456  name = var_info['obs_name']
457  extra = var_info['obs_extra']
458  elif data_type == 'ENS':
459  level = var_info['ens_level']
460  thresh = var_info['ens_thresh']
461  name = var_info['ens_name']
462  extra = var_info['ens_extra']
463  else:
464  return ''
465 
466  # check if python embedding is used and set up correctly
467  # set env var for file type if it is used
468  pyEmbedIsOK = self.check_for_python_embedding(data_type, var_info)
469  if not pyEmbedIsOK:
470  return ''
471 
472  next_field = self.get_field_info(v_level=level,
473  v_thresh=thresh,
474  v_name=name,
475  v_extra=extra,
476  d_type=data_type)
477  if next_field is None:
478  return ''
479 
480  field_list.extend(next_field)
481 
482  return ','.join(field_list)

485  def find_model_members(self, time_info):
486  """! Finds the model member files to compare
487  Args:
488  @param time_info dictionary containing timing information
489  @rtype string
490  @return Returns a list of the paths to the ensemble model files
491  """
492  model_dir = self.c_dict['FCST_INPUT_DIR']
493  # used for filling in missing files to ensure ens_thresh check is accurate
494  fake_dir = '/ensemble/member/is/missing'
495 
496  # model_template is a list of 1 or more.
497  ens_members_path = []
498 
499  # get all files that exist
500  for ens_member_template in self.c_dict['FCST_INPUT_TEMPLATE']:
501  member_file = do_string_sub(ens_member_template,
502  **time_info)
503  expected_path = os.path.join(model_dir, member_file)
504 
505  # if wildcard expression, get all files that match
506  if '?' in expected_path or '*' in expected_path:
507  wildcard_files = sorted(glob.glob(expected_path))
508  self.logger.debug('Ensemble members file pattern: {}'
509  .format(expected_path))
510  self.logger.debug('{} members match file pattern'
511  .format(str(len(wildcard_files))))
512 
513  # add files to list of ensemble members
514  for wildcard_file in wildcard_files:
515  ens_members_path.append(wildcard_file)
516  else:
517  # otherwise check if file exists
518  expected_path = util.preprocess_file(expected_path,
519  self.c_dict['FCST_INPUT_DATATYPE'],
520  self.config)
521 
522  # if the file exists, add it to the list
523  if expected_path:
524  ens_members_path.append(expected_path)
525  else:
526  # add relative path to fake dir and add to list
527  ens_members_path.append(os.path.join(fake_dir, member_file))
528  self.logger.warning('Expected ensemble file {} not found'
529  .format(member_file))
530 
531  # if more files found than expected, error and exit
532  if len(ens_members_path) > self.c_dict['N_MEMBERS']:
533  msg = 'Found more files than expected! ' +\
534  'Found {} expected {}. '.format(len(ens_members_path),
535  self.c_dict['N_MEMBERS']) +\
536  'Adjust wildcard expression in [filename_templates] '+\
537  'FCST_ENSEMBLE_STAT_INPUT_TEMPLATE or adjust [config] '+\
538  'ENSEMBLE_STAT_N_MEMBERS. Files found: {}'.format(ens_members_path)
539  self.log_error(msg)
540  self.logger.error("Could not file files in {} for init {} f{} "
541  .format(model_dir, time_info['init_fmt'],
542  str(time_info['lead_hours'])))
543  return False
544  # if fewer files found than expected, warn and add fake files
545  elif len(ens_members_path) < self.c_dict['N_MEMBERS']:
546  msg = 'Found fewer files than expected. '+\
547  'Found {} expected {}.'.format(len(ens_members_path),
548  self.c_dict['N_MEMBERS'])
549  self.logger.warning(msg)
550  # add fake files to list to get correct number of files for ens_thresh
551  diff = self.c_dict['N_MEMBERS'] - len(ens_members_path)
552  self.logger.warning('Adding {} fake files to '.format(str(diff))+\
553  'ensure ens_thresh check is accurate')
554  for _ in range(0, diff, 1):
555  ens_members_path.append(fake_dir)
556 
557  # write file that contains list of ensemble files
558  list_filename = time_info['init_fmt'] + '_' + \
559  str(time_info['lead_hours']) + '_ensemble.txt'
560  return self.write_list_file(list_filename, ens_members_path)
