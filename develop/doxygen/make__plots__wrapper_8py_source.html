237  def get_met_version(self):
238  stat_analysis_exe = os.path.join(self.config.getdir('MET_BIN_DIR'),
239  'stat_analysis')
240  p = subprocess.Popen([stat_analysis_exe, "--version"],
241  stdout=subprocess.PIPE)
242  out, err = p.communicate()
243  out = out.decode(encoding='utf-8', errors='strict')
244  for line in out.split('\n'):
245  if 'MET Version:' in line:
246  met_verison_line = line
247  met_version_str = (
248  met_verison_line.partition('MET Version:')[2].split('V')[1]
249  )
250  if len(met_version_str) == 3:
251  met_version = float(met_version_str)
252  else:
253  met_version = float(met_version_str.rpartition('.')[0])
