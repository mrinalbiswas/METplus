149  def reformat_grid_id(self, grid_id):
150  """!Reformat the grid id (MASK_GRID value in the configuration
151  file) if it starts with G. Looks for G<n> where n is a
152  digit 0-999 and zero pads the value. i.e. G7 becomes G007,
153  G13 becomes G013, etc.
154  Args:
155  @param grid_id identifier of grid
156  Returns:
157  @return reformatted grid id if valid, None if not
158  """
159  # If grid ID does not start with G, return it
160  if not grid_id.startswith('G'):
161  return grid_id
162 
163  # look for G<n> where n is a digit 0-999
164  match = re.match(r'G([0-9]{1,3})$', grid_id)
165 
166  # pad with zeros if found
167  if match:
168  number = match.group(1)
169  return 'G' + number.zfill(3)
170 
171  # Unexpected format
172  self.log_error('Grid id in unexpected format of Gn or ' +
173  'Gnn, please check again. Exiting...')
174  return None

294  def get_command(self):
295  """! Builds the command to run the MET application
296  @rtype string
297  @return Returns a MET command with arguments that you can run
298  """
299  if self.app_path is None:
300  self.log_error('No app path specified. You must use a subclass')
301  return None
302 
303  cmd = '{} -v {} '.format(self.app_path, self.c_dict['VERBOSITY'])
304 
305  for a in self.args:
306  cmd += a + " "
307 
308  if len(self.infiles) == 0:
309  self.log_error("No input filenames specified")
310  return None
311 
312  # if multiple input files, add first now, then add rest with
313  # -pbfile argument
314  cmd += self.infiles[0] + " "
315 
316  if self.outfile == "":
317  self.log_error("No output filename specified")
318  return None
319 
320  if self.outdir == "":
321  self.log_error("No output directory specified")
322  return None
323 
324  out_path = os.path.join(self.outdir, self.outfile)
325 
326  # create outdir (including subdir in outfile) if it doesn't exist
327  if not os.path.exists(os.path.dirname(out_path)):
328  os.makedirs(os.path.dirname(out_path))
329 
330  cmd += out_path + ' '
331 
332  cmd += self.c_dict['CONFIG_FILE'] + ' '
333 
334  if len(self.infiles) > 1:
335  for f in self.infiles[1:]:
336  cmd += '-pbfile ' + f + ' '
337 
338  if self.c_dict['VALID_WINDOW_BEGIN']:
339  cmd += f"-valid_beg {self.c_dict['VALID_WINDOW_BEGIN']} "
340 
341  if self.c_dict['VALID_WINDOW_END']:
342  cmd += f"-valid_end {self.c_dict['VALID_WINDOW_END']} "
343 
344  return cmd.strip()
