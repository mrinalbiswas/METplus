465  def process_data(self, basin, cyclone, model_list, time_info):
466  """!Find requested files and run tc_pairs
467  Args:
468  @param basin region of storm from config
469  @param cyclone ID number of cyclone from config
470  @param model_list list of models that be available
471  @param time_info object containing timing information to process
472  """
473 
474  # set regex expressions for basin and cyclone if wildcard is used
475  # cast cyclone value to integer if it is not a wildcard
476  if cyclone != '*':
477  cyclone_regex = cyclone
478  else:
479  cyclone_regex = "([0-9]{2,4})"
480 
481  if basin != '??':
482  basin_regex = basin
483  else:
484  basin_regex = "([a-zA-Z]{2})"
485 
486  # get search expression for bdeck files to pass to glob
487  string_sub = do_string_sub(self.c_dict['BDECK_TEMPLATE'],
488  basin=basin,
489  cyclone=cyclone,
490  **time_info)
491  bdeck_glob = os.path.join(self.c_dict['BDECK_DIR'],
492  string_sub)
493  self.logger.debug('Looking for BDECK: {}'.format(bdeck_glob))
494 
495  # get all files that match expression
496  bdeck_files = sorted(glob.glob(bdeck_glob))
497 
498  # if no bdeck_files found
499  if len(bdeck_files) == 0:
500  template = self.c_dict['BDECK_TEMPLATE']
501  self.log_error(f'No BDECK files found searching for basin {basin} '
502  f'and cyclone {cyclone} using template {template}')
503  return False
504 
505  # find corresponding adeck or edeck files
506  for bdeck_file in bdeck_files:
507  self.logger.debug('Found BDECK: {}'.format(bdeck_file))
508 
509  # set current basin and cyclone from bdeck file
510  # if basin or cyclone are a wildcard, these will be
511  # replaced by the value pulled from the bdeck file
512  current_basin = basin
513  current_cyclone = cyclone
514 
515  # if wildcard was used in bdeck, pull out what was
516  # substituted for * to find corresponding bdeck file
517  matches = []
518  if '*' in bdeck_glob or '?' in bdeck_glob:
519  # get regex expression to pull out basin and cyclone
520  string_sub = do_string_sub(self.c_dict['BDECK_TEMPLATE'],
521  basin=basin_regex,
522  cyclone=cyclone_regex,
523  **time_info)
524  bdeck_regex = os.path.join(self.c_dict['BDECK_DIR'],
525  string_sub)
526 
527  # capture wildcard values in template
528  bdeck_regex = bdeck_regex.replace('*', '(.*)')
529  bdeck_regex = bdeck_regex.replace('?', '(.)')
530 
531  match = re.match(bdeck_regex, bdeck_file)
532  if match:
533  matches = match.groups()
534  tags = get_tags(self.c_dict['BDECK_TEMPLATE'])
535  match_count = 0
536  for tag in tags:
537  # if wildcard is set for tag found, get value
538  # if wildcard if found in template, increment index
539  if tag == 'basin' and basin == '??':
540  current_basin = matches[match_count]
541  match_count += 1
542  elif tag == 'cyclone' and cyclone == '*':
543  current_cyclone = matches[match_count]
544  match_count += 1
545  elif tag == '*' or tag == '?':
546  match_count += 1
547 
548  # create lists for deck files, put bdeck in list so it can be handled
549  # the same as a and e for reformatting even though it will always be
550  # size 1
551  bdeck_list = [bdeck_file]
552  adeck_list = []
553  edeck_list = []
554 
555  # get adeck files
556  if self.c_dict['GET_ADECK']:
557  adeck_list = self.find_deck_files('A', current_basin,
558  current_cyclone, model_list,
559  time_info)
560  # get edeck files
561  if self.c_dict['GET_EDECK']:
562  edeck_list = self.find_deck_files('E', current_basin,
563  current_cyclone, model_list,
564  time_info)
565 
566  if not adeck_list and not edeck_list:
567  self.log_error('Could not find any corresponding '
568  'ADECK or EDECK files')
569  continue
570 
571  # reformat extra tropical cyclone files if necessary
572  if self.c_dict['REFORMAT_DECK']:

727  Args:
728  @param in_csvfile input csv file that is being parsed

747

764  # Delete the third column
